11 #include <map>
12 #include <vector>
13 
14 using namespace ccgl;
15 using std::map;
16 using std::vector;
17 
18 class TaskInfo: Interface {
19 public:
20  /*! Constructor
21  * \param[in] size Number of process
22  * \param[in] rank Rank ID
23  */
24  explicit TaskInfo(int size, int rank);
25  ~TaskInfo();
26  /// Check global variables
27  bool CheckInputData();
28  /// Build various data structures to assist calculation
29  bool Build();
30  /// Malloc space for transferred values
31  void MallocTransferredValues(int transfer_count, int multiplier);
32  /// Get the number of subbasins in current rank
33  int GetSubbasinNumber();
34  /// Get the maximum layering ID in current rank
35  int GetMaxLayerID() { return max_lyr_; }
36  int GetGlobalMaxLayerID() { return max_lyr_all_; }
37  vector<int>& GetRankSubbasinIDs() { return rank_subbsn_id_; }
38  map<int, int>& GetSubbasinRank() { return subbsn_rank_; }
39  map<int, int>& GetSubbasinLayer() { return subbsn_layer_; }
40  map<int, int>& GetDownstreamID() { return downstream_; }
41  map<int, vector<int> >& GetUpstreamIDs() { return upstreams_; }
42  map<int, bool>& GetUpstreamsInRank() { return upstreams_inrank_; }
43  map<int, vector<int> >& GetLayerSubbasinIDs() { return lyr_subbsns_; }
44  map<int, vector<int> >& GetSourceLayerSubbasinIDs() { return srclyr_subbsns_; }
45  map<int, vector<int> >& GetNonSourceLayerSubbasinIDs() { return nonsrclyr_subbsns_; }
46  map<int, map<int, float *> >& GetSubbasinTransferredValues() { return subbsn_tfvalues_; }
47  map<int, map<int, float *> >& GetReceivedSubbasinTransferredValues() { return recv_subbsn_tfvalues_; }
48 public:
49  int max_len; ///< Max. subbasins number of all tasks
50  int subbsn_count; ///< All subbasins number
51  int* subbsn_id; ///< Subbasin IDs in all groups, length: max_len * size_
52  int* lyr_id; ///< Layering number of each subbasins, length: max_len * size_
53  int* down_id; ///< Down stream subbasin ID of each subbasin, length: max_len * size_
54  int* up_count; ///< Upstream subbasin numbers of each subbasin, length: max_len * size_
55  int* up_ids; ///< Upstream subbasin IDs of each subbasin, length: max_len * size_ * MAX_UPSTREAM
56 
57 private:
58  int size_; ///< Number of process
59  int rank_; ///< Rank ID
60  int* subbsn_count_rank_; ///< Subbasin number in each rank
61  int max_lyr_; ///< Max. layering number of current rank
62  int max_lyr_all_; ///< Global max. layering number
63  vector<int> rank_subbsn_id_; ///< Subbasin IDs in current rank
64  /*! Subbasin object -> rank ID (i.e., group ID)
65  * Key: Subbasin ID of the whole basin
66  * Value: rank ID
67  */
68  map<int, int> subbsn_rank_;
69  /*! Subbasin object -> layering ID
70  * Key: Subbasin ID of the whole basin
71  * Value: layering ID
72  */
73  map<int, int> subbsn_layer_;
74  /*! Downstream subbasin ID
75  * Key: Subbasin ID of the whole basin
76  * Value: Downstream subbasin ID
77  */
78  map<int, int> downstream_;
79  /*! Upstream subbasins ID
80  * Key: Subbasin ID of the whole basin
81  * Value: Upstream subbasins ID
82  */
83  map<int, vector<int> > upstreams_;
84  /*! If upstream subbasins in the same rank
85  * Key: Subbasin ID of the whole basin
86  * Value: true or false
87  */
88  map<int, bool> upstreams_inrank_;
89  /*! Subbasins of each layer of current rank
90  * Key: Layering ID
91  * Value: Source subbasin IDs in current rank
92  */
93  map<int, vector<int> > lyr_subbsns_;
94  /*! Source subbasins of each layer of current rank
95  * Key: Layering ID
96  * Value: Source subbasin IDs in current rank
97  */
98  map<int, vector<int> > srclyr_subbsns_;
99  /*! Non source subbasins in each layer in current slave rank
100  * Key: Layering ID
101  * Value: Non source subbasin IDs in current rank
102  */
103  map<int, vector<int> > nonsrclyr_subbsns_;
104  /*! Transferred values of subbasins in current rank with timestep stamp
105  * Key: Timestep sequence, which is equal to layer ID in numerical.
106  * Value: Transferred values of subbasins, in which key is subbasinID and value is transferred values
107  */
108  map<int, map<int, float *> > subbsn_tfvalues_;
109  /*! Received transferred values of subbasins in current rank with timestep stamp
110  * Key: Timestep sequence, which is equal to layer ID in numerical.
111  * Value: Transferred values of subbasins, in which key is subbasinID and value is transferred values
112  */
113  map<int, map<int, float *> > recv_subbsn_tfvalues_;
114 };
115 #endif /* SEIMS_MPI_TASK_INFO_H */
